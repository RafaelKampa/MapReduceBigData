package tde1.exercicio6;

import org.apache.hadoop.io.Writable;
import org.apache.hadoop.io.WritableComparable;
import tde1.exercicio5.GroupByCommdtAvgBrazil;

import java.io.DataInput;
import java.io.DataOutput;
import java.io.IOException;

public class MaxMinAvgWritable implements Writable {
    private int n;
    private Double price;
    private Double min;
    private Double max;

    public MaxMinAvgWritable(){

    }

    public MaxMinAvgWritable(int n, Double price, Double min, Double max) {
        this.n = n;
        this.price = price;
        this.min = min;
        this.max = max;
    }

    @Override
    public void readFields(DataInput in) throws IOException {
        price = in.readDouble();
        n = in.readInt();
        min = in.readDouble();
        max = in.readDouble();
    }

    @Override
    public void write(DataOutput out) throws  IOException {
        out.writeDouble(price);
        out.writeInt(n);
        out.writeDouble(min);
        out.writeDouble(max);
    }

    public int getN(){
        return this.n;
    }

    public void setN(){
        this.n = n;
    }

    public void setN(int n) {
        this.n = n;
    }

    public Double getPrice() {
        return price;
    }

    public void setPrice(Double price) {
        this.price = price;
    }

    public Double getMin() {
        return min;
    }

    public void setMin(Double min) {
        if (getMin() < getPrice()) {
            this.min = min;
        } else {
            this.min = getPrice();
        }
    }

    public Double getMax() {
        return max;
    }

    public void setMax(Double max) {
        if (getMax() > getPrice()){
            this.max = max;
        } else {
            this.max = getPrice();
        }
    }

    @Override
    public String toString() {
        return " Avg price = " + price + " - Min = " + min + " - Max = " + max;
    }

}
